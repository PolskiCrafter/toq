			check_variable = { num_owned_states > 1 }

		fire_only_once = yes
	
		cost = 50
		
		complete_effect = {
			if = {
				limit = {
					has_government = gemefraism
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@gemefraism
				}
			}
			if = {
				limit = {
					has_government = kollektigleichenism
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@kollektigleichenism
				}
			}
			if = {
				limit = {
					has_government = ubestrapanism
				}
				set_variable = {
					var = civil_war_size_var
					value = party_popularity@ubestrapanism
				}
			}
			subtract_from_variable = {
				var = civil_war_size_var
				value = army_support_var
			}
			if = {
				limit = {
					check_variable = {
						var = civil_war_size_var
						value = 0.3
						compare = less_than
					}
				}
				set_variable = {
					var = civil_war_size_var
					value = 0.3
				}
			}
			start_civil_war = {
				ruling_party = futhdeomism
				ideology = ROOT
				size = civil_war_size_var
				keep_unit_leaders_trigger = {
					has_trait = hidden_sympathies
				}
			}
			clr_country_flag = preparation_for_futhdeomism_civil_war
			clr_country_flag = military_support_for_futhdeomism_civil_war
			clr_country_flag = civil_support_for_futhdeomism_civil_war
			set_country_flag = ideology_civil_war
		}
		
		ai_will_do = {
			factor = 0
		}
	}
